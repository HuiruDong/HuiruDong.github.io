多种继承方式

前端 继承
JS 混淆知识点
发布日期:   2020-10-30

这是一篇笔记的笔记，偶然间在 github 发现了 这个 repo，其中深入系列还是不错的，至少讲的比较明白，适合我这样的孩子。把之前自己不太清楚的地方搞搞清楚。

本篇主要是整理一下多种继承方式，面试的时候总问到继承。在看继承之前还是推荐把原型和原型链搞搞清楚，不然看着有点晕。

原型链继承

最基本的继承，也是比较好理解的。宗旨就是 Child 的原型指向 Parent 的实例。

function Parent() {
    this.name = 'parent'
}
Parent.prototype.say = function() { console.log('Hi~') }

function Child() {}
Child.prototype = new Parent()

let c = new Child()
c.name // parent
c.say() // Hi~

这种继承方式会有两个很严重的问题：

  • 引用类型的值会被共享
  • 在创建 Child 的实例（也就是 new Child）的时候不能向 Parent 传值

我们先看第一个问题，注意说的是引用类型的值。

function Parent() {
    this.colors = {
        a: 'green'
    }
}

function Child() {}
Child.prototype = new Parent()

let c = new Child()
let c1 = new Child()
c.colors.a = 'golden'

c.colors.a // golden
c1.colors.a // golden

按常理来说每一个实例应该都是独立的，与其它实例互不干涉，但是在原型链继承中引用值类型是被共用的。顺着我们看第二个问题，就算是 new Child 的时候传值了，怎么给 Parent 是不是个问题。

借用构造函数（经典继承）

通过 call 或者 apply 把 Parent 放在 Child 里面。

function Parent() {
    this.name = 'Eva'
    this.colors = ['pink', 'green']
}

function Child() {
    Parent.call(this)
}

let c =  new Child()
c.name  // Eva
c.colors  // ['pink', 'green']

我们来看看有没有解决原型链继承的两个问题：引用类型值被共享；不能传参。

function Parent() {
    this.name = 'Eva'
    this.colors = ['pink', 'green']
}

function Child() {
    Parent.call(this)
}

let c =  new Child()
let c1 = new Child()
c.colors[0] = 'blue'

c.colors  // ["blue", "green"]
c1.colors  // ["pink", "green"]

看来共享的问题解决了，保证了每个 Child 实例都是独立的。

传参问题也是 OK 的，毕竟 call() 和 apply() 是可以接收参数的，只是接受方式不太一样。

function Parent(name) {
    this.name = name
}

function Child(name) {
    Parent.call(this, name)
}

let c =  new Child('Alice')
let c1 = new Child('Eva')

c.name  // Alice
c1.name // Eva

不过缺点是方法是在构造函数中定义的，导致每次创建实例都会创建一遍方法。

组合继承

原型链继承 + 构造函数继承

function Parent(name) {
    this.name = name
}

Parent.prototype.say = function() {
    console.log(`I\'m ${this.name}`)
}

function Child(name) {
    // 构造函数继承
    Parent.call(this, name)
}
// 原型链继承
Child.prototype = new Parent()
/**
 * 这里需要注意一个事情
 * 当我们将原型指到 Parent 的实例身上，那么 Child 的原型指向的构造函数就变成了 Parent
 * 这是原型链继承的副作用
 * 我们需要指回来
 **/
Child.prototype.constructor = Child

let c1 = new Child('Alice')
c1.say()  // I'm Alice

这种继承方式的优点在于解决了传参问题和引用类型值的共享问题。

function Parent(name) {
    this.name = name
    this.colors = ['blue', 'green']
}

function Child(name) {
    Parent.call(this, name)
}

Child.prototype = new Parent()
Child.prototype.constructor = Child

let c1 = new Child('Alice')
let c2 = new Child('Bonnie')

c1.colors[0] = 'pink'

c1.colors  // ['pink', 'green']
c2.colors // ['blue', 'green']

原型式继承

将传入的对象作为创建的对象的原型。

其实特别像是“引擎自动关联一个对象给你”一样，只是这里我们手动关联。

这也是 Object.create() 的模拟实现

function createObj(o) {
    function F() {}
    F.prototype = o
    return new F()
}

let obj = {
    name: 'Bonnie'
}

let o1 = createObj(obj)

它的问题在于引用类型会被共享

function createObj(o) {
    function F() {}
    F.prototype = o
    return new F()
}

let obj = {
    colors: ['Bonnie']
}

let o1 = createObj(obj)
let o2 = createObj(obj)

o1.colors[0] = 'User'
o2.colors // ['User']

寄生式继承

创建一个仅用于封装继承过程的函数，该函数在内部以某种形式来做增强对象，最后返回对象。

function createObj(o) {
    let clone = Object.create(o)
    clone.say = function() {
        console.log('Hi~')
    }
    return clone
}

let obj = {}

let o1 = createObj(obj)

这种继承方式的问题在于

  • 引用值类型共享
  • 每次创建都会创建一遍方法，这一点和经典继承一样
function createObj(o) {
    let clone = Object.create(o)
    clone.say = function() {
        console.log('Hi~')
    }
    return clone
}

let obj = {
    colors: ['Bonnie']
}

let o1 = createObj(obj)
let o2 = createObj(obj)

o1.colors[0] = 'User'
o2.colors // ['User']
